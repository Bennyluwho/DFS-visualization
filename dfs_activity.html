Depth-First Search (DFS) — Pseudocode

  procedure DFS(G, start):
    visited ← set()
    parent  ← map()
    time    ← 0

    function visit(u):
        visited.add(u)
        time ← time + 1
        for v in G.neighbors(u):
            if v ∉ visited:
                parent[v] ← u
                visit(v)
        time ← time + 1

    visit(start)
    return parent
  
  procedure DFS_iter(G, start):
    visited ← set()
    parent  ← map()
    S ← stack([start])

    while S not empty:
        u ← S.pop()
        if u ∉ visited:
            visited.add(u)
            for v in reverse(G.neighbors(u)):
                if v ∉ visited:
                    parent[v] ← u
                    S.push(v)
    return parent
