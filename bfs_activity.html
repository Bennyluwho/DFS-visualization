def bfs(graph, start):
    visited = set()
    queue = []   # blank #1
    while queue:
        node = queue.pop(0)   # dequeue from front
        if node not in visited:
            visited.add(node)
            # enqueue neighbors so they are explored in order
            queue.extend()
    return visited
